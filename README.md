ئاسانترین ڕێگا بۆ بەکارهێنانی  lorem ipsum بۆ پیتی کوردی و لاتینی


# Installation and usage
# دامەزراندن و بەکار‌هێنان

const univGenerate = require('univGenerate');

const theText = univGenerate('ku','w',2);
